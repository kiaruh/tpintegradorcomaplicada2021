cantidad=$1 #Numero ingreado por teclado

aux=1

aux2=1

if [ $1 -gt 1 ]

then

echo $aux2

fi

for((i=3;i<=cantidad;i++))

do

val=`expr $aux + $aux2`

echo $val

aux=$aux2

aux2=$val

done

}
